-- prints nodeno, maxX, maxY, minX, minY (for a mbr of rectangles in nodeno)
SELECT SRid.nodeno, MAX(S.maxX), MAX(S.maxY), MIN(S.minX), MIN(S.minY)
FROM square_rowid SRid, square S
WHERE SRid.rowid = S.id
GROUP BY SRid.nodeno
ORDER BY SRid.nodeno DESC;
